```go
func (cm *ConnectionManager) GetTenantDBByUUID(schoolUUID guuid.UUID) (*gorm.DB, error) {
    var school models.SchoolConnection
    if err := cm.coreDB.Where("uuid = ?", schoolUUID).First(&school).Error; err != nil {
        return nil, fmt.Errorf("school with uuid %s not found: %w", schoolUUID, err)
    }
    return cm.GetTenantDB(school.ID)
}
```
Note: This queries the `schools` table by uuid, but SchoolConnection doesn't have a UUID field. The query still works (GORM maps `Where("uuid = ?", ...)` to the actual column), but the UUID won't be scanned into SchoolConnection.

       ```go
       // VerifyUUIDsExist checks that all rows in the given table have non-null UUIDs.
       // Call after migration to ensure data integrity.
       func VerifyUUIDsExist(db *gorm.DB, tableName string) error {
           var count int64
           err := db.Table(tableName).
               Where("uuid IS NULL").
               Count(&count).Error
           if err != nil {
               return fmt.Errorf("verify UUIDs for %s: %w", tableName, err)
           }
           if count > 0 {
               return fmt.Errorf("%d rows in %s have NULL UUIDs", count, tableName)
           }
           return nil
       }
       ```
