  <action>
    Refactor `ProvisionSchool()` to use a "rollback on failure" pattern via deferred cleanup.

    **Approach**: Rather than wrapping the entire flow in a PostgreSQL transaction (which cannot contain DDL like CREATE SCHEMA without special handling), use a **compensating action** pattern:

    ```go
    func (s *ProvisioningService) ProvisionSchool(ctx context.Context, schoolID, adminUserID uint, adminFirstname, adminLastname, adminPassport string) (err error) {
        schemaName := fmt.Sprintf("school_%d", schoolID)

        // ── Rollback guard ───────────────────────────────────────
        // Track which steps completed so we can undo them on failure.
        var schemaCreated bool
        var schemaNameUpdated bool
        var migrationsRun bool
        var adminUserInfoCreated bool
        var seedsCompleted bool

        // Deferred rollback: if err != nil at exit, undo each completed step
        // in reverse order.
        defer func() {
            if r := recover(); r != nil {
                err = fmt.Errorf("PANIC in ProvisionSchool: %v", r)
            }
            if err == nil {
                return // success — no rollback needed
            }
            logger.Warnf("[provisioning] Rolling back provisioning for school %d schema %q: %v", schoolID, schemaName, err)

            // Rollback in reverse order
            if seedsCompleted {
                // No-op: seeds are idempotent and re-running is cheaper than cleanup
            }
            if adminUserInfoCreated {
                // Delete UserInfo record
                adminDB := NewSchemaDB(s.coreDB, schemaName).DB()
                if delErr := adminDB.Where("user_id = ?", adminUserID).Delete(&models.UserInfo{}).Error; delErr != nil {
                    logger.Warnf("[provisioning] rollback: failed to delete admin UserInfo: %v", delErr)
                }
            }
            if migrationsRun {
                // Drop and recreate schema with migrations tracking table
                // This effectively resets the schema for a fresh retry.
                // We DON'T drop the schema here — the schema DROP is the final
                // step that also covers migrations cleanup.
            }
            if schemaNameUpdated {
                if resetErr := s.coreDB.Table("schools").Where("id = ?", schoolID).
                    Update("schema_name", "").Error; resetErr != nil {
                    logger.Warnf("[provisioning] rollback: failed to reset schema_name: %v", resetErr)
                }
            }
            if schemaCreated {
                rawDB, rawErr := s.coreDB.DB()
                if rawErr == nil {
                    if _, dropErr := rawDB.ExecContext(context.Background(), "DROP SCHEMA IF EXISTS "+schemaName+" CASCADE"); dropErr != nil {
                        logger.Warnf("[provisioning] rollback: failed to drop schema %q: %v", schemaName, dropErr)
                    } else {
                        logger.Infof("[provisioning] Rolled back: dropped schema %q", schemaName)
                    }
                }
            }
            // Reset database_status to pending
            if statusErr := s.coreDB.Table("schools").Where("id = ?", schoolID).
                Update("database_status", "pending").Error; statusErr != nil {
                logger.Warnf("[provisioning] rollback: failed to reset database_status: %v", statusErr)
            }
        }()

        // ── Existing provisioning steps: set flags as each completes ──
        // Step 1: CREATE SCHEMA
        rawDB, err := s.coreDB.DB()
        if err != nil { return }
        if _, err := rawDB.ExecContext(ctx, "CREATE SCHEMA IF NOT EXISTS "+schemaName); err != nil { return }
        schemaCreated = true

        // Step 2: Store schema_name
        if err := s.coreDB.WithContext(ctx).Table("schools").
            Where("id = ?", schoolID).
            Updates(map[string]interface{}{"schema_name": schemaName}).Error; err != nil { return }
        schemaNameUpdated = true

        // Step 3: Run migrations
        if err := s.migrationSvc.ApplySchoolMigrationsForSchema(s.coreDB, schemaName, schoolID); err != nil { return }
        migrationsRun = true

        // Step 4: Create admin UserInfo
        // ... (existing code, set adminUserInfoCreated = true on success)

        // Step 5: Seed default academic data
        // ... (existing code, set seedsCompleted = true on success)

        // Step 6: Mark school as active
        // ... (existing code)
    }
    ```

    **Key design decisions**:
    - Use **compensating actions** (deferred rollback) instead of wrapping in a single PG transaction — DDL (CREATE SCHEMA) cannot run inside a transaction that also does DML without special tx_ddl handling.
    - **DROP SCHEMA ... CASCADE** as the nuclear option on failure — it cleanly removes everything created during provisioning.
    - The `defer` checks `err != nil` — if the function returns success, no rollback occurs.
    - Panics are caught by `recover()` and converted to errors, triggering the same rollback path.

    **Edge cases**:
    - **Idempotent schema creation**: `CREATE SCHEMA IF NOT EXISTS` — safe to call on re-provision.
    - **Schema already exists with active status**: detect this early and return nil (idempotent).
    - **ProvisioningFailed status**: treat as fresh provisioning — drop schema if exists, re-run everything.
    - **Recover() + rollback**: if a panic occurs, the recover sets `err`, the deferred rollback runs. Must use named return `(err error)` for this to work.

    **2a: Create** `backend/internal/database/tenant/provisioning_recovery.go`:

    ```go
    package tenant

    import (
        "context"
        "fmt"

        "github.com/playbits/schoolcare-v2/internal/database/models"
        "github.com/playbits/schoolcare-v2/pkg/logger"
    )

    // RecoveryService handles re-provisioning of schools stuck in a failed
    // provisioning state.
    type RecoveryService struct {
        coreDB            *gorm.DB
        provisioningSvc   *ProvisioningService
    }

    func NewRecoveryService(coreDB *gorm.DB, provisioningSvc *ProvisioningService) *RecoveryService {
        return &RecoveryService{
            coreDB:          coreDB,
            provisioningSvc: provisioningSvc,
        }
    }

    // ReprovisionSchool re-provisions a school that is in provisioning_failed
    // status. It drops the existing schema (if any) and re-runs the full
    // provisioning pipeline.
    func (s *RecoveryService) ReprovisionSchool(ctx context.Context, schoolID uint, adminUserID uint, adminFirstname, adminLastname, adminPassport string) error {
        // Look up the school
        var school models.School
        if err := s.coreDB.First(&school, schoolID).Error; err != nil {
            return fmt.Errorf("school %d not found: %w", schoolID, err)
        }

        // If school is active, refuse re-provision
        if school.DatabaseStatus == "active" {
            return fmt.Errorf("school %d is already active - cannot re-provision", schoolID)
        }

        // Drop existing schema if it exists (from failed provisioning)
        if school.SchemaName != nil && *school.SchemaName != "" {
            rawDB, err := s.coreDB.DB()
            if err != nil {
                return fmt.Errorf("get raw DB: %w", err)
            }
            if _, err := rawDB.ExecContext(ctx, "DROP SCHEMA IF EXISTS "+*school.SchemaName+" CASCADE"); err != nil {
                logger.Warnf("[recovery] failed to drop existing schema %q: %v", *school.SchemaName, err)
                // Continue — ProvisionSchool handles IF NOT EXISTS
            }
            logger.Infof("[recovery] Dropped existing schema %q for school %d", *school.SchemaName, schoolID)
        }

        // Run full provisioning
        return s.provisioningSvc.ProvisionSchool(ctx, schoolID, adminUserID, adminFirstname, adminLastname, adminPassport)
    }
    ```
