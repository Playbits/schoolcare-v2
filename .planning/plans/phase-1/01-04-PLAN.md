    ```go
    package pdf

    import (
        "bytes"
        "fmt"
        "io"
        "mime/multipart"
        "net/http"
    )

    type GotenbergClient struct {
        baseURL    string
        httpClient *http.Client
        htmlGen    *HTMLGenerator
    }

    func NewGotenbergClient(baseURL string) *GotenbergClient {
        return &GotenbergClient{
            baseURL:    baseURL,
            httpClient: &http.Client{Timeout: 30 * time.Second},
            htmlGen:    NewHTMLGenerator(),
        }
    }

    // GenerateHTML delegates to the embedded HTMLGenerator.
    func (c *GotenbergClient) GenerateHTML(data ReportCardData) ([]byte, error) {
        return c.htmlGen.GenerateHTML(data)
    }

    // GeneratePDF sends HTML to Gotenberg and returns PDF bytes.
    // POST /forms/chromium/convert/html with multipart form containing the HTML file.
    func (c *GotenbergClient) GeneratePDF(html []byte) ([]byte, error) {
        var buf bytes.Buffer
        w := multipart.NewWriter(&buf)

        // Create form file field
        fw, err := w.CreateFormFile("files", "report.html")
        if err != nil {
            return nil, fmt.Errorf("create form file: %w", err)
        }
        if _, err := fw.Write(html); err != nil {
            return nil, fmt.Errorf("write html to form: %w", err)
        }

        // Set paper size to A4
        if err := w.WriteField("paperSize", "A4"); err != nil {
            return nil, fmt.Errorf("write paper size: %w", err)
        }
        // Margins (in inches)
        if err := w.WriteField("marginTop", "0.5"); err != nil {
            return nil, fmt.Errorf("write margin top: %w", err)
        }
        if err := w.WriteField("marginBottom", "0.5"); err != nil {
            return nil, fmt.Errorf("write margin bottom: %w", err)
        }
        if err := w.WriteField("marginLeft", "0.5"); err != nil {
            return nil, fmt.Errorf("write margin left: %w", err)
        }
        if err := w.WriteField("marginRight", "0.5"); err != nil {
            return nil, fmt.Errorf("write margin right: %w", err)
        }
        w.Close()

        req, err := http.NewRequest("POST", c.baseURL+"/forms/chromium/convert/html", &buf)
        if err != nil {
            return nil, fmt.Errorf("create request: %w", err)
        }
        req.Header.Set("Content-Type", w.FormDataContentType())

        resp, err := c.httpClient.Do(req)
        if err != nil {
            return nil, fmt.Errorf("gotenberg request: %w", err)
        }
        defer resp.Body.Close()

        if resp.StatusCode != http.StatusOK {
            body, _ := io.ReadAll(resp.Body)
            return nil, fmt.Errorf("gotenberg returned status %d: %s", resp.StatusCode, string(body))
        }

        pdfBytes, err := io.ReadAll(resp.Body)
        if err != nil {
            return nil, fmt.Errorf("read gotenberg response: %w", err)
        }
        return pdfBytes, nil
    }
    ```

    ```go
    func (s *ReportCardService) buildSubjectData(ctx context.Context, studentID uint, session, term string) []pdf.SubjectData {
        // Query grade_item_scores for the student, joined with grade_items,
        // assessments, and subjects to build subject data.
        //
        // Schema: scores are stored in grade_item_scores table with
        // grade_item_id, student_id, score_value, max_score.
        // Grade items belong to assessments, assessments belong to curriculums,
        // subjects are linked to grade items via curriculum structure.
        //
        // For MVP, query grade_item_scores for this student, group by subject,
        // compute totals and percentages.

        var results []pdf.SubjectData

        // Check if tenant DB has the scores table
        type ScoreRow struct {
            SubjectName string
            Score       float64
            MaxScore    float64
        }

        var rows []ScoreRow
        // Query scores from grade_item_scores joined through the curriculum
        // structure to get subject names.
        // This is a simplified query — actual join structure depends on the
        // curriculum/assessment/grade_item/scores model relationships.
        if err := s.db.WithContext(ctx).
            Table("grade_item_scores").
            Select("subjects.name as subject_name, SUM(grade_item_scores.score_value) as score, SUM(grade_item_scores.max_score) as max_score").
            Joins("JOIN grade_items ON grade_items.id = grade_item_scores.grade_item_id").
            Joins("JOIN assessments ON assessments.id = grade_items.assessment_id").
            Joins("JOIN curriculums ON curriculums.id = assessments.curriculum_id").
            Joins("JOIN subjects ON subjects.id = curriculums.subject_id").
            Where("grade_item_scores.student_id = ? AND grade_item_scores.session = ? AND grade_item_scores.term = ?", studentID, session, term).
            Group("subjects.name").
            Find(&rows).Error; err != nil {
            logger.Warnf("[reportcard] buildSubjectData query failed: %v", err)
            return results
        }

        for _, row := range rows {
            grade, gradePoint := pdf.ScoreToGrade(row.Score)
            results = append(results, pdf.SubjectData{
                Name:       row.SubjectName,
                Score:      row.Score,
                MaxScore:   row.MaxScore,
                Grade:      grade,
                GradePoint: gradePoint,
            })
        }

        return results
    }
    ```

    ```go
    func (h *ReportCardHandler) DownloadReportCard(c *gin.Context) {
        idStr := c.Param("id")
        id, err := strconv.ParseUint(idStr, 10, 64)
        if err != nil {
            helpers.SendError(c, appErr.NewBadRequestError("invalid report card id"))
            return
        }

        rc, err := h.service.GetByID(c.Request.Context(), uint(id))
        if err != nil {
            helpers.SendError(c, appErr.NewNotFoundError("report card not found"))
            return
        }

        // Generate PDF on-the-fly (or retrieve from cache/URL if already generated)
        tenantDB := middleware.GetTenantDB(c)
        pdfBytes, err := h.service.GeneratePDF(c.Request.Context(), tenantDB, uint(id))
        if err != nil {
            helpers.SendError(c, appErr.NewInternalError("failed to generate PDF"))
            return
        }

        c.Data(200, "application/pdf", pdfBytes)
    }
    ```
