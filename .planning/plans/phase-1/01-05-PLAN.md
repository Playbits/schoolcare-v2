    import "time"

    // SchedulerConfig defines cron schedule expressions for periodic jobs.
    // Each field is a standard 5-field cron expression (min hour dom mon dow).
    type SchedulerConfig struct {
        // BackupCron: when to run nightly tenant backups. Default: 2:00 AM daily.
        BackupCron string

        // ReportCron: when to run weekly report generation. Default: 3:00 AM Monday.
        ReportCron string

        // BillingCron: when to run monthly billing. Default: 4:00 AM on the 1st.
        BillingCron string

        // Location for cron timezone. Defaults to UTC.
        Location *time.Location
    }

    // DefaultSchedulerConfig returns sensible defaults for all cron schedules.
    func DefaultSchedulerConfig() SchedulerConfig {
        return SchedulerConfig{
            BackupCron:  "0 2 * * *",      // daily at 2:00 AM
            ReportCron:  "0 3 * * 1",      // weekly on Monday at 3:00 AM
            BillingCron: "0 4 1 * *",      // monthly on the 1st at 4:00 AM
            Location:    time.UTC,
        }
    }

    **1b: Create** `backend/internal/scheduler/cron_runner.go`:

    ```go
    package scheduler

    import (
        "github.com/robfig/cron/v3"
        "github.com/playbits/schoolcare-v2/pkg/logger"
    )

    // CronRunner manages periodic job scheduling using robfig/cron/v3.
    // All jobs delegate to Asynq for fault-tolerant execution.
    type CronRunner struct {
        cron       *cron.Cron
        entryIDs   []cron.EntryID
    }

    // NewCronRunner creates a CronRunner with the given config and job functions.
    func NewCronRunner(cfg SchedulerConfig, jobs *Jobs) *CronRunner {
        opts := []cron.Option{
            cron.WithLocation(cfg.Location),
            cron.WithLogger(cron.VerbosePrintfLogger(logger.NewCronWriter())), // adapter if needed
            cron.WithChain(
                cron.SkipIfStillRunning(cron.VerbosePrintfLogger(logger.NewCronWriter())),
                cron.Recover(cron.VerbosePrintfLogger(logger.NewCronWriter())),
            ),
        }

        c := cron.New(opts...)

        var entryIDs []cron.EntryID

        if jobs.NightlyBackup != nil {
            id, err := c.AddFunc(cfg.BackupCron, jobs.NightlyBackup)
            if err != nil {
                logger.Warnf("[scheduler] failed to register backup job: %v", err)
            } else {
                entryIDs = append(entryIDs, id)
                logger.Infof("[scheduler] registered backup job: %s", cfg.BackupCron)
            }
        }

        if jobs.WeeklyReport != nil {
            id, err := c.AddFunc(cfg.ReportCron, jobs.WeeklyReport)
            if err != nil {
                logger.Warnf("[scheduler] failed to register report job: %v", err)
            } else {
                entryIDs = append(entryIDs, id)
                logger.Infof("[scheduler] registered report job: %s", cfg.ReportCron)
            }
        }

        if jobs.MonthlyBilling != nil {
            id, err := c.AddFunc(cfg.BillingCron, jobs.MonthlyBilling)
            if err != nil {
                logger.Warnf("[scheduler] failed to register billing job: %v", err)
            } else {
                entryIDs = append(entryIDs, id)
                logger.Infof("[scheduler] registered billing job: %s", cfg.BillingCron)
            }
        }

        return &CronRunner{
            cron:     c,
            entryIDs: entryIDs,
        }
    }

    // Start begins executing scheduled jobs in a background goroutine.
    func (r *CronRunner) Start() {
        r.cron.Start()
        logger.Infof("[scheduler] Cron runner started with %d jobs", len(r.entryIDs))
    }

    // Stop gracefully shuts down the scheduler, waiting for running jobs.
    // Returns the context after all running jobs complete or ctx is cancelled.
    func (r *CronRunner) Stop() {
        logger.Info("[scheduler] Stopping cron runner...")
        ctx := r.cron.Stop()
        <-ctx.Done()
        logger.Info("[scheduler] Cron runner stopped")
    }
    ```

    package scheduler

    import (
        "context"

        "github.com/hibiken/asynq"
        "gorm.io/gorm"

        "github.com/playbits/schoolcare-v2/internal/queue"
        "github.com/playbits/schoolcare-v2/pkg/logger"
    )

    // Jobs groups all periodic job functions. Each job enqueues Asynq tasks.
    // Dependencies (DB, queue client) are injected at construction time.
    type Jobs struct {
        coreDB        *gorm.DB
        queueClient   *queue.QueueClient

        // Job functions — set by NewJobs, called by CronRunner
        NightlyBackup func()
        WeeklyReport  func()
        MonthlyBilling func()
    }

    // NewJobs creates job functions with injected dependencies.
    func NewJobs(coreDB *gorm.DB, queueClient *queue.QueueClient) *Jobs {
        j := &Jobs{
            coreDB:      coreDB,
            queueClient: queueClient,
        }
        j.NightlyBackup = j.nightlyBackupJob
        j.WeeklyReport = j.weeklyReportJob
        j.MonthlyBilling = j.monthlyBillingJob
        return j
    }

    // nightlyBackupJob enqueues a backup:create task for every active school.
    func (j *Jobs) nightlyBackupJob() {
        ctx := context.Background()

        // Find all active schools with schemas
        type schoolRow struct{ ID uint }
        var schools []schoolRow
        if err := j.coreDB.Table("schools").
            Select("id").
            Where("database_status = 'active' AND schema_name IS NOT NULL AND schema_name != ''").
            Find(&schools).Error; err != nil {
            logger.Errorf("[scheduler] nightly backup: list schools: %v", err)
            return
        }

        for _, school := range schools {
            task, err := queue.NewBackupCreateTask(school.ID)
            if err != nil {
                logger.Warnf("[scheduler] nightly backup: create task for school %d: %v", school.ID, err)
                continue
            }
            if err := j.queueClient.Enqueue(task, asynq.MaxRetry(3)); err != nil {
                logger.Warnf("[scheduler] nightly backup: enqueue for school %d: %v", school.ID, err)
            }
        }

        logger.Infof("[scheduler] nightly backup: enqueued %d backup tasks", len(schools))
    }

    // weeklyReportJob enqueues report:generate tasks (placeholder — reports
    // need school/class context; expand when report card batch logic matures).

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | E | CronRunner.Start/Stop | mitigate | Graceful shutdown via cron.Cron.Stop() — waits for running jobs; server uses defer pattern |
| T-05-02 | D | Job panics | mitigate | cron.Recover chain logs panic and continues; job-level error handling wraps panics in deferred recover |
| T-05-03 | E | Concurrent job overlap | mitigate | cron.SkipIfStillRunning chain prevents a second instance if the first is still running (e.g., if nightly backup takes >24h) |
| T-05-04 | I | Missing Asynq = no scheduling | accept | CronRunner gracefully degrades: if queueClient is nil, jobs are nil-safe (log warning, no crash) |
| T-05-05 | E | Backup job enumerates all schools | mitigate | Query uses indexed fields (database_status, schema_name); reasonable for expected schema count (<500) |
</threat_model>
