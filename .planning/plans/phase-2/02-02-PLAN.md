    // ── BehaviorIncidentRepository ────────────────────────────

    type BehaviorIncidentRepository struct {
        db *gorm.DB
    }

    func NewBehaviorIncidentRepository(db *gorm.DB) *BehaviorIncidentRepository {
        return &BehaviorIncidentRepository{db: db}
    }

    func (r *BehaviorIncidentRepository) Create(ctx context.Context, incident *models.BehaviorIncident) error {
        return r.db.WithContext(ctx).Create(incident).Error
    }

    func (r *BehaviorIncidentRepository) FindByID(ctx context.Context, id uint) (*models.BehaviorIncident, error) {
        var incident models.BehaviorIncident
        err := r.db.WithContext(ctx).Preload("Student").First(&incident, id).Error
        if err != nil {
            return nil, err
        }
        return &incident, nil
    }

    func (r *BehaviorIncidentRepository) FindBySchool(ctx context.Context, schoolID uint, q IncidentListQuery) ([]models.BehaviorIncident, int64, error) {
        // Query via Student join to scope by school
        query := r.db.WithContext(ctx).

    // calcGrade implements the conduct grading rubric.
    // A: zero incidents
    // B: 1-2 incidents, all mild
    // C: 3-5 incidents, all ≤ moderate
    // D: 6+ incidents or any severe (without critical)
    // F: any critical incident, or any suspension
    func (s *DisciplineService) calcGrade(totalIncidents, totalDetentions, totalSuspensions int, incidents []models.BehaviorIncident) string {
        if totalIncidents == 0 {
            return "A"
        }

        hasCritical := false
        hasSevere := false
        hasModerate := false
        for _, inc := range incidents {
            switch inc.SeverityLevel {
            case "critical":
                hasCritical = true
            case "severe":
                hasSevere = true
            case "moderate":
                hasModerate = true
            }
        }

        if hasCritical || totalSuspensions > 0 {
            return "F"
        }
        if hasSevere || totalIncidents >= 6 {
            return "D"
        }
        if hasModerate || totalIncidents >= 3 {
            return "C"
        }
        return "B"
    }

    func (s *DisciplineService) GetStudentSummary(ctx context.Context, tenantDB *gorm.DB, schoolID, studentID uint) (*StudentDisciplineSummary, error) {

        suspensionRepo := s.suspensionRepo

            conductRepo = NewConductRecordRepository(tenantDB)
            detentionRepo = NewDetentionRepository(tenantDB)
            suspensionRepo = NewSuspensionRepository(tenantDB)
        }

        incidents, _, _ := incidentRepo.FindBySchool(ctx, schoolID, IncidentListQuery{StudentID: studentID, Limit: 100})
        conductHistory, _ := conductRepo.FindByStudent(ctx, studentID)
        totalDetentions, _ := detentionRepo.CountByStudent(ctx, studentID)
        totalSuspensions, _ := suspensionRepo.CountByStudent(ctx, studentID)

        // Build response
        recentIncidents := make([]IncidentResponse, 0, len(incidents))
        for _, inc := range incidents {
            recentIncidents = append(recentIncidents, toIncidentResponse(&inc))
        }

        conductResponses := make([]ConductRecordResponse, len(conductHistory))
        for i, cr := range conductHistory {
            conductResponses[i] = toConductRecordResponse(&cr)
        }

        return &StudentDisciplineSummary{
            StudentID:       studentID,
            TotalIncidents:  len(incidents),
            TotalDetentions: totalDetentions,
            TotalSuspensions: totalSuspensions,
            ConductHistory:  conductResponses,
            RecentIncidents: recentIncidents,
