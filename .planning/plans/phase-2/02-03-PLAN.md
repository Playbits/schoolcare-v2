    ```go
    package grading

    import (
        "context"
        "gorm.io/gorm"
        "github.com/playbits/schoolcare-v2/internal/database/models"
    )

    type GradingRepositoryInterface interface {
        FindBoundariesBySchool(ctx context.Context, schoolID uint) ([]models.GradeBoundary, error)
        UpsertBoundaries(ctx context.Context, boundaries []models.GradeBoundary) error
        DeleteBoundariesBySchool(ctx context.Context, schoolID uint, excludeIDs []uint) error
    }

    type GradingRepository struct {
        db *gorm.DB
    }

    func NewGradingRepository(db *gorm.DB) *GradingRepository {
        return &GradingRepository{db: db}
    }

    func (r *GradingRepository) FindBoundariesBySchool(ctx context.Context, schoolID uint) ([]models.GradeBoundary, error) {
        var boundaries []models.GradeBoundary
        err := r.db.WithContext(ctx).Where("school_id = ?", schoolID).Order("min_percentage DESC").Find(&boundaries).Error
        return boundaries, err
    }

    // UpsertBoundaries creates or updates grade boundaries.
    // For each boundary, if ID is zero it's created; otherwise updated.
    func (r *GradingRepository) UpsertBoundaries(ctx context.Context, boundaries []models.GradeBoundary) error {
        for i := range boundaries {
            if boundaries[i].ID > 0 {
                if err := r.db.WithContext(ctx).Save(&boundaries[i]).Error; err != nil {
                    return err
                }
            } else {
                if err := r.db.WithContext(ctx).Create(&boundaries[i]).Error; err != nil {
                    return err
                }
            }
        }
        return nil
    }

    // DeleteBoundariesBySchool deletes all boundaries for a school except those in excludeIDs.
    func (r *GradingRepository) DeleteBoundariesBySchool(ctx context.Context, schoolID uint, excludeIDs []uint) error {
        return r.db.WithContext(ctx).
            Where("school_id = ? AND id NOT IN ?", schoolID, excludeIDs).
            Delete(&models.GradeBoundary{}).Error
    }
    ```

    ```go
    package grading

    import (
        "context"
        "encoding/json"
        "sort"

        "gorm.io/gorm"

        domerr "github.com/playbits/schoolcare-v2/internal/errors"
        "github.com/playbits/schoolcare-v2/internal/database/models"
    )

    // DefaultWAECBoundaries returns the standard WAEC A1-F9 grading scale.
    // Used when no school-specific or session-specific overrides exist.
    var DefaultWAECBoundaries = []models.GradeBoundary{
        {Grade: "A1", MinPercentage: 75, MaxPercentage: 100, IsCreditPass: true, GradePoint: 1, Description: "Excellent"},
        {Grade: "B2", MinPercentage: 70, MaxPercentage: 74, IsCreditPass: true, GradePoint: 2, Description: "Very Good"},
        {Grade: "B3", MinPercentage: 65, MaxPercentage: 69, IsCreditPass: true, GradePoint: 3, Description: "Good"},
        {Grade: "C4", MinPercentage: 60, MaxPercentage: 64, IsCreditPass: true, GradePoint: 4, Description: "Credit"},
        {Grade: "C5", MinPercentage: 55, MaxPercentage: 59, IsCreditPass: true, GradePoint: 5, Description: "Credit"},
        {Grade: "C6", MinPercentage: 50, MaxPercentage: 54, IsCreditPass: true, GradePoint: 6, Description: "Credit"},
        {Grade: "D7", MinPercentage: 45, MaxPercentage: 49, IsCreditPass: false, GradePoint: 7, Description: "Pass"},
        {Grade: "E8", MinPercentage: 40, MaxPercentage: 44, IsCreditPass: false, GradePoint: 8, Description: "Pass"},
        {Grade: "F9", MinPercentage: 0, MaxPercentage: 39, IsCreditPass: false, GradePoint: 9, Description: "Fail"},
    }

    // GradingService handles grade boundary management and grade computation.
    type GradingService struct {
        repo GradingRepositoryInterface
    }

    func NewGradingService(repo GradingRepositoryInterface) *GradingService {
        return &GradingService{repo: repo}
    }

    // GetActiveBoundaries returns the effective grade boundaries for a school.
    // Priority: session-level config > school-level DB rows > DefaultWAECBoundaries.
    func (s *GradingService) GetActiveBoundaries(ctx context.Context, tenantDB *gorm.DB, schoolID uint, session *models.Session) ([]models.GradeBoundary, error) {
        // 1. Check session-level GradeScaleConfig first
        if session != nil && session.GradeScaleConfig != nil {
            var sessionBoundaries []models.GradeBoundary
            if err := json.Unmarshal(session.GradeScaleConfig, &sessionBoundaries); err == nil && len(sessionBoundaries) > 0 {
                return sessionBoundaries, nil
            }
        }

        // 2. Check school-level boundaries in DB
        dbBoundaries, err := s.repo.FindBoundariesBySchool(ctx, schoolID)
        if err != nil || len(dbBoundaries) > 0 {
            return dbBoundaries, nil
        }

        // 3. Fall back to default
        result := make([]models.GradeBoundary, len(DefaultWAECBoundaries))
        copy(result, DefaultWAECBoundaries)
        return result, nil
    }

    // ComputeGrade maps a percentage score to a grade letter using the given boundaries.
    // Boundaries must be sorted by MinPercentage descending (highest first).
    // Percentage is clamped to [0, 100].
    func ComputeGrade(percentage float32, boundaries []models.GradeBoundary) string {
        if percentage < 0 {
            percentage = 0
        }
        if percentage > 100 {
            percentage = 100
        }

        // Sort descending by MinPercentage so we match highest applicable first
        sorted := make([]models.GradeBoundary, len(boundaries))
        copy(sorted, boundaries)
        sort.Slice(sorted, func(i, j int) bool {
            return sorted[i].MinPercentage > sorted[j].MinPercentage
        })

        for _, b := range sorted {
            if percentage >= b.MinPercentage && percentage <= b.MaxPercentage {
                return b.Grade
            }
        }
        return "F9" // fallback
    }

    // ComputeCreditPass returns true if the grade is a credit pass (A1-C6).
    func ComputeCreditPass(grade string, boundaries []models.GradeBoundary) bool {
        for _, b := range boundaries {
            if b.Grade == grade {
                return b.IsCreditPass
            }
        }
        return false
    }

    // ComputeGradePoint returns the grade point for a given grade letter.
    func ComputeGradePoint(grade string, boundaries []models.GradeBoundary) int {
        for _, b := range boundaries {
            if b.Grade == grade {
                return b.GradePoint
            }
        }
        return 0
    }

    // GetGradeBoundaries handles GET /grading/grade-boundaries
    func (s *GradingService) GetGradeBoundaries(ctx context.Context, tenantDB *gorm.DB, schoolID uint, sessionID uint) ([]GradeBoundaryResponse, error) {
        var session *models.Session
        if sessionID > 0 {
            if err := tenantDB.WithContext(ctx).First(&session, sessionID).Error; err != nil {
                return nil, domerr.NewNotFoundError("session not found")
            }
        }

        boundaries, err := s.GetActiveBoundaries(ctx, tenantDB, schoolID, session)
        if err != nil {
            return nil, err
        }

        resp := make([]GradeBoundaryResponse, len(boundaries))
        for i, b := range boundaries {
            resp[i] = GradeBoundaryResponse{
                ID:            b.ID,
                Grade:         b.Grade,
                MinPercentage: b.MinPercentage,
                MaxPercentage: b.MaxPercentage,
                IsCreditPass:  b.IsCreditPass,
                GradePoint:    b.GradePoint,
                Description:   b.Description,
            }
        }
        return resp, nil
    }

    // UpdateGradeBoundaries handles PUT /grading/grade-boundaries
    func (s *GradingService) UpdateGradeBoundaries(ctx context.Context, tenantDB *gorm.DB, schoolID uint, req *UpdateGradeBoundariesRequest) ([]GradeBoundaryResponse, error) {
        // Validate ratios within each input
        for i, input := range req.Boundaries {
            if input.MinPercentage > input.MaxPercentage {
                return nil, domerr.NewValidationError(
                    "boundary " + input.Grade + ": min_percentage must be <= max_percentage")
            }
            _ = i
        }

        // Build model slice
        models := make([]models.GradeBoundary, len(req.Boundaries))
        for i, input := range req.Boundaries {
            m := models.GradeBoundary{
                SchoolID:      schoolID,
                Grade:         input.Grade,
                MinPercentage: input.MinPercentage,
                MaxPercentage: input.MaxPercentage,
                IsCreditPass:  input.IsCreditPass,
                GradePoint:    input.GradePoint,
                Description:   input.Description,
            }
            if input.ID != nil {
                m.ID = *input.ID
            }
            models[i] = m
        }

        if err := s.repo.UpsertBoundaries(ctx, models); err != nil {
            return nil, err
        }

        // Collect IDs of upserted boundaries
        var ids []uint
        for _, m := range models {
            if m.ID > 0 {
                ids = append(ids, m.ID)
            }
    ```

    ```go
    // ──────────────────────────────────────────────────────────────
    // CA/Exam split computation
    // ──────────────────────────────────────────────────────────────

    // CABreakdown is the internal result of splitting scores by CA/Exam type.
    type CABreakdown struct {
        CAScore      float32
        CATotal      float32
        ExamScore    float32
        ExamTotal    float32
        Total        float32
        Grade        string
        IsCreditPass bool
        GradePoint   int
    }

    // ComputeCAExamSplit calculates the weighted CA and Exam totals for
    // a student's scores across all assessments in a curriculum.
    //
    // Algorithm:
    //   1. Fetch all assessments for the curriculum (via repository)
    //   2. Separate into CA vs Exam based on IsCA/IsExam flags
    //   3. Sum raw CA scores / sum CA max totals → CA raw percentage
    //   4. Sum raw Exam scores / sum Exam max totals → Exam raw percentage
    //   5. Final = CA_raw% × ca_ratio + Exam_raw% × exam_ratio
    //   6. Map final percentage to grade boundary
    //
    // When no assessments are classified (all IsCA=false, IsExam=false),
    // returns nil — no split to apply, caller uses existing raw score.
    func (s *ScoreService) ComputeCAExamSplit(
        ctx context.Context,
        tenantDB *gorm.DB,
        schoolID uint,
        studentID uint,
        subjectID uint,
        sessionID uint,
        curriculumID uint,
        caRatio float32,
        examRatio float32,
    ) (*CABreakdown, error) {
        tempAcademicRepo, tempScoreRepo, _, _, _ := s.reposForTenant(tenantDB)

        // Fetch assessments for the curriculum
        assessments, err := tempAcademicRepo.FindAssessmentsByCurriculum(ctx, curriculumID)
        if err != nil {
            return nil, fmt.Errorf("fetch assessments: %w", err)
        }

        // Separate into CA and Exam
        var caAssessments, examAssessments []models.Assessment
        hasClassification := false
        for _, a := range assessments {
            if a.IsCA {
                caAssessments = append(caAssessments, a)
                hasClassification = true
            }
            if a.IsExam {
                examAssessments = append(examAssessments, a)
                hasClassification = true
            }
        }

        // If no assessments are classified, return nil (no split)
        if !hasClassification {
            return nil, nil
        }

        // Sum CA scores
        var caRawTotal, caMaxTotal float32
        for _, ca := range caAssessments {
            score, err := tempScoreRepo.FindScoreByCompositeKey(ctx, ca.ID, sessionID, 0, subjectID, studentID)
            if err != nil || score == nil {
                continue // missing CA score — skip (student may not have all CAs yet)
            }
            raw := s.parseScoreTotal(score.Score)
            caRawTotal += raw
            caMaxTotal += ca.Total
        }

        // Sum Exam scores
        var examRawTotal, examMaxTotal float32
        for _, exam := range examAssessments {
            score, err := tempScoreRepo.FindScoreByCompositeKey(ctx, exam.ID, sessionID, 0, subjectID, studentID)
            if err != nil || score == nil {
                continue // missing exam score — skip
            }
            raw := s.parseScoreTotal(score.Score)
            examRawTotal += raw
            examMaxTotal += exam.Total
        }

        // Compute weighted total
        var caPercent, examPercent float32
        if caMaxTotal > 0 {
            caPercent = (caRawTotal / caMaxTotal) * 100
        }
        if examMaxTotal > 0 {
            examPercent = (examRawTotal / examMaxTotal) * 100
        }
        finalPercent := (caPercent * caRatio) + (examPercent * examRatio)

        // Clamp to [0, 100]
        if finalPercent < 0 {
            finalPercent = 0
        }
        if finalPercent > 100 {
            finalPercent = 100
        }

        // Map to grade using the grading service
        gradingService := NewGradingService(NewGradingRepository(tenantDB))
        boundaries, err := gradingService.GetActiveBoundaries(ctx, tenantDB, schoolID, nil)
        if err != nil {
            return nil, fmt.Errorf("get grade boundaries: %w", err)
        }

        grade := ComputeGrade(finalPercent, boundaries)
        isCreditPass := ComputeCreditPass(grade, boundaries)
        gradePoint := ComputeGradePoint(grade, boundaries)

        // Scale CA and Exam back to ratio-weighted values for display
        weightedCATotal := caRatio * 100
        if weightedCATotal < 0 {
            weightedCATotal = 0
        }
        weightedExamTotal := examRatio * 100
        if weightedExamTotal < 0 {
            weightedExamTotal = 0
        }

        weightedCAScore := (caPercent / 100) * weightedCATotal
        weightedExamScore := (examPercent / 100) * weightedExamTotal

        return &CABreakdown{
            CAScore:      roundFloat32(weightedCAScore, 1),
            CATotal:      roundFloat32(weightedCATotal, 1),
            ExamScore:    roundFloat32(weightedExamScore, 1),
            ExamTotal:    roundFloat32(weightedExamTotal, 1),
            Total:        roundFloat32(finalPercent, 1),
            Grade:        grade,
            IsCreditPass: isCreditPass,
            GradePoint:   gradePoint,
        }, nil
    }

    // parseScoreTotal extracts the "total" field from the score JSON blob.
    // Returns 0 if no total is set (student hasn't been graded yet).
    func (s *ScoreService) parseScoreTotal(scoreJSON datatypes.JSON) float32 {
        if scoreJSON == nil {
            return 0
        }
        var data map[string]interface{}
        if err := json.Unmarshal(scoreJSON, &data); err != nil {
            return 0
        }
        if t, ok := data["total"].(float64); ok {
            return float32(t)
        }
        return 0
    }

    // roundFloat32 rounds a float32 to the given number of decimal places.
    func roundFloat32(val float32, decimals int) float32 {
        pow := float32(math.Pow(10, float64(decimals)))
        return float32(math.Round(float64(val)*float64(pow))) / pow
    }
    ```
