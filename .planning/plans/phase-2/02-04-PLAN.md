<objective>
Build the NGN Fee Structure Configuration subsystem: models for Nigerian fee items, per-class/term fee structures, waivers at student level, a debtors list with outstanding balance breakdown, automated fee reminders via the Communication module, and partial payment tracking.

**Purpose**: Allow school admins to define common Nigerian fee items (tuition, development levy, PTA, exam, sports, uniform, transport, etc.), assign them per class/term with NGN amounts, configure waivers (full/partial), view a debtors list with breakdowns, trigger fee reminders, and track partial payments against individual fee items.

**Output**: Production-grade fee configuration CRUD endpoints, debtor aggregation queries, reminder dispatch integration, and partial payment ledger — all within the existing finance module at `backend/internal/modules/finance/`.
</objective>

    func computeFeeItemStatus(
        feeStructure models.FeeStructure,
        totalPaid float64,
        waivers []models.FeeWaiver,
    ) (effectiveDue float64, status string) {
        effectiveDue = feeStructure.Amount
        for _, w := range waivers {
            if w.FeeStructureID == feeStructure.ID {
                if time.Now().Before(*w.ExpiresAt) || w.ExpiresAt == nil { // waiver active
                    if w.WaiverType == "full" {
                        effectiveDue = 0
                    } else if w.WaiverType == "partial" {
                        effectiveDue = feeStructure.Amount * (1 - w.Percentage/100)
                    }
                }
            }
        }
        if totalPaid >= effectiveDue {
            return effectiveDue, "paid"
        } else if totalPaid > 0 {
            return effectiveDue, "partial"
        }
        return effectiveDue, "unpaid"
    }
